# Test to excercise phi generation when the variables have order dependancies

main

procedure depTest;
var x,y,z;
{
  let x <- call InputNum();
  let y <- x;
  let z <- 0;
  # x and y will have a dependancy between them, the order of moves
  # needs to ensure that the phi semantics is preserved
  while z == 0 do
    let x <- y;
    let y <- y + 1;
    let z <- 1
  od;
  call OutputNum(x);
  call OutputNum(y)
};

{
  call depTest
}.
